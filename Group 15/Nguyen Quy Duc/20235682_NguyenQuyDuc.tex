\chapter{Nguyên lý thiết kế và mẫu thiết kế áp dụng}
\section{Nguyên lý thiết kế}
\begin{itemize}
    \item \textbf{Single Responsibility Principle (SRP)}: Tách biệt logic quản lý hồ sơ Site khỏi UI. `SiteService` đảm nhận vai trò điều phối lưu trữ và gửi thông báo, còn tầng hiển thị chỉ tiếp nhận đầu vào của người dùng.
    \item \textbf{High Cohesion}: Đóng gói toàn bộ các thực thể liên quan đến thông tin đối tác nước ngoài vào mô hình `ImportSite` giúp cấu trúc domain chặt chẽ, dễ kiểm tra và phát triển.
\end{itemize}

\section{Mẫu thiết kế}
\begin{itemize}
    \item \textbf{Facade}: Lớp `ImportOrderingFacade` tích hợp và điều phối cuộc gọi nghiệp vụ của hệ thống.
    \item \textbf{Repository (Port and Adapter)}: Quản lý độc lập dữ liệu các đối tác Site.
\end{itemize}

\subsection{Thiết kế kiểm thử hộp trắng (White-box Testing - Độ bao phủ C1)}
Bao phủ các nhánh kiểm tra ràng buộc trước khi lưu hồ sơ Site:
\begin{itemize}
    \item \textbf{Nhánh 1}: Ngày vận chuyển bằng tàu hoặc bằng máy bay nhỏ hơn hoặc bằng 0 $\rightarrow$ Ném `BusinessException`.
    \item \textbf{Nhánh 2}: Danh mục chứa mặt hàng không tồn tại $\rightarrow$ Ném `BusinessException`.
    \item \textbf{Nhánh 3}: Dữ liệu hợp lệ hoàn toàn $\rightarrow$ Lưu thông tin Site mới và gửi thông báo thay đổi thành công.
\end{itemize}

package vn.edu.hust.itss.group15.usecases;

import static org.junit.jupiter.api.Assertions.assertEquals;
import static org.junit.jupiter.api.Assertions.assertThrows;
import static org.junit.jupiter.api.Assertions.assertTrue;

import java.util.Set;
import org.junit.jupiter.api.Test;
import vn.edu.hust.itss.group15.application.BusinessException;
import vn.edu.hust.itss.group15.application.ImportOrderingFacade;
import vn.edu.hust.itss.group15.infrastructure.persistence.memory.InMemoryStore;

class Uc002SiteProfileTest {
  @Test
  void tcUc00201UpdatesValidSiteProfile() {
    ImportOrderingFacade facade = facade();

    var site = facade.sites().updateSite("SITE-SEA-01", "Singapore Fast Site", 12, 3, Set.of("MH-001", "MH-002"));

    assertEquals("Singapore Fast Site", site.siteName());
    assertEquals(12, site.deliveryDaysByShip());
    assertEquals(3, site.deliveryDaysByAir());
    assertTrue(site.merchandiseCatalog().contains("MH-002"));
  }

  @Test
  void tcUc00202RejectsInvalidTransitTimeOrCatalog() {
    ImportOrderingFacade facade = facade();

    assertThrows(BusinessException.class, () -> facade.sites().updateSite("SITE-SEA-01", "Site", 0, 5, Set.of("MH-001")));
    assertThrows(BusinessException.class, () -> facade.sites().updateSite("SITE-SEA-01", "Site", 10, 5, Set.of("UNKNOWN")));
  }

  private ImportOrderingFacade facade() {
    return new ImportOrderingFacade(new InMemoryStore());
  }
}
